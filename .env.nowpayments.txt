# Add these to your .env file:

NOWPAYMENTS_API_KEY=your_actual_api_key_here
NOWPAYMENTS_IPN_SECRET=your_actual_ipn_secret_here

# Steps to get your API key:
# 1. Go to https://nowpayments.io/
# 2. Sign up or log in
# 3. Go to Settings > API Keys
# 4. Copy your API key
# 5. Copy your IPN secret (for webhook verification)
# 6. Replace the values above in your .env file
# 7. Run: php artisan config:clear

# For testing, you can use NOWPayments Sandbox:
# https://sandbox.nowpayments.io/
# Get sandbox API key from sandbox dashboard
